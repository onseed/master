var pdSearch = {
	set : function(num, setSlecter , arrayTxt, arrayObj, arraySet){//카드 생성		contbox
		arraySet.txtCall = arrayTxt[num][3];//문구
		arraySet.leftCall = arrayTxt[num][4];//왼쪽 버튼
		arraySet.leftNum = arrayTxt[num][1];//왼쪽 버튼 넘버링
		arraySet.rightCall = arrayTxt[num][5];//오른쪽 버튼명			
		arraySet.rightNum = arrayTxt[num][2];//오른쪽 버튼 넘버링
		arraySet.backNum = arrayTxt[num][7];//백넘버 
		arraySet.imgCall =  arrayTxt[num][6];//이미지 클래스명: q0~q(n)

		$el = $(setSlecter);
		$quest = $el.find('.quest');
		$btn = $el.find('.faq_btn .btn_cont');
		$leftBtn = $el.find('.faq_btn .btn_cont.left');
		$rightBtn = $el.find('.faq_btn .btn_cont.right');						
		$quest.html(arraySet.txtCall);			
		$leftBtn.text(arraySet.leftCall);
		$rightBtn.text(arraySet.rightCall);
		$(setSlecter+' .pdcar .car').addClass(arraySet.imgCall);
		pdSearch.click(num, setSlecter, arrayTxt, arrayObj, arraySet);
		
		console.log("set:"+setSlecter);
	},
	init: function(opt){//selector: class name
		var selector = opt.name;
		var arrayTxt = opt.arrayFaq; 
		var arrayObj = opt.arrayPd;
		var arraySet = opt.arraySet;
		
		if(isMobile){$(selector).addClass('mobile');}//모바일일때						
		pdSearch.set(0, selector+' .contbox', arrayTxt, arrayObj, arraySet);//세팅
		$(selector+' .btn_sean').attr('disabled',true);//다시,이전 버튼 막기
		
		if(arraySet.sliderOpt == true){//***** 슬라이더 처리 ******
			var selHeight = $(selector).find('.contbox').outerHeight();//box 높이 
			$(selector).addClass("faq_slider").css({'height':selHeight});//box 높이 처리				
			
			cloneHtml= '
슬라이더선택
';
			arraySet.clonedata = cloneHtml; 				
			
			$(selector).append(arraySet.clonedata);					
			$(selector+' .contbox').eq(0).addClass('current');
			$(selector+' .contbox.clone').css({"left" : $(selector).width(), "display":"none"});								
		}//*****// 슬라이더 처리 ******
		
		$(selector+' .btn_sean').click(function(e){
			$(selector+' .contbox .faq_btn').fadeIn(arraySet.aniSpeed);
			$(selector+' .contbox .result_btn, .contbox .pdtext .pd ').css('display','none');
			$(selector+' .contbox .pdcar .car').removeClass(arraySet.imgCall);
			
			if($(this).hasClass('retry')){ //'다시하기'일때
				pdSearch.set(0, selector+' .contbox', arrayTxt, arrayObj, arraySet);	
			}else{//'이전으로'일때
				pdSearch.set(arraySet.backNum, selector+' .contbox', arrayTxt, arrayObj, arraySet);	
			}
			if(arraySet.imgCall=='q0'){
				$(selector+' .btn_sean').attr('disabled',true);//다시,이전 버튼 막기
			}
			arraySet.returnClick = false;
			
			$(selector+' .contbox').css('display','none');
			$(selector+' .contbox').stop().fadeIn(arraySet.aniSpeed,function(){
				arraySet.returnClick = true;
			});
		});			
		
	},		
	click : function(num, setSlecter, arrayTxt, arrayObj, arraySet){			
		$btn.off().on('click',function(e){
			var selWd = $(setSlecter).width();//셀렉터 넓이
			if(arraySet.returnClick){//click useble					
				$(setSlecter+ ' .pdcar .car').removeClass(arraySet.imgCall);//이전 이미지
				if($(this).hasClass('left')){//left button
					arraySet.seanNxt = arraySet.leftNum;
				}else{//right button
					arraySet.seanNxt = arraySet.rightNum;
				}
				arraySet.returnClick = false;
				//console.log('backNum: '+arraySet.backNum+', seanNxt: '+arraySet.seanNxt);
				
				if(arraySet.seanNxt>10){//seanNxt이10이상은 결과
					return pdSearch.result(arraySet.seanNxt, setSlecter , arrayTxt, arrayObj, arraySet);
				}					
				
				if(arraySet.sliderOpt == true){//***** 슬라이더 처리 ******
					
					var cloneSelecter;
					if(setSlecter.indexOf(".current") > -1){ //커런트 여부
						setSlecter = setSlecter;
					}else{
						setSlecter = setSlecter+".current";
					}
					cloneSelecter =  setSlecter.replace(".current",".clone");
					
					arraySet.clonedata = $(setSlecter+'.current').html(); //console.log(arraySet.clonedata);				
				
					$(cloneSelecter).html('');
					$(cloneSelecter).html(arraySet.clonedata);
				}//*****// 슬라이더 처리 ******
				
				pdSearch.set(arraySet.seanNxt,setSlecter, arrayTxt, arrayObj, arraySet);
									
				if(arraySet.sliderOpt == true){//***** 슬라이더 처리 ******
					var currentHeight = $(setSlecter).outerHeight();//box 높이
					$(this).closest('.faq_slider').css({'height':currentHeight});////box 높이 처리
					
					if($(this).hasClass('left')){//왼쪽버튼인경우 왼쪽에서 들어온다 
						$(setSlecter).css({"margin-left" : -selWd});
						$(cloneSelecter).css({"margin-left" : -selWd, "display":"block"});
						$(setSlecter+','+ cloneSelecter).animate({"margin-left" : 0},arraySet.sliderSpeed,function(){
							$(cloneSelecter).css({"display":"none"});
							arraySet.returnClick = true;
						});
					}else{//오른쪽버튼인경우 오른쪽에서 들어온다
						$(setSlecter).css({"margin-left" : selWd});
						$(cloneSelecter).css({"margin-left" : -selWd, "display":"block"});							
						$(setSlecter).animate({"margin-left" : 0},arraySet.sliderSpeed,function(){								
							arraySet.returnClick = true;
						});
						$(cloneSelecter).animate({"margin-left" : -selWd*2},arraySet.sliderSpeed,function(){
							$(cloneSelecter).css({"margin-left" : 0, "display":"none"});
						});
					}
				}//*****// 슬라이더 처리 ******					
				else{
					$(setSlecter).css('display','none');
					$(setSlecter).fadeIn(arraySet.aniSpeed,function(){
						arraySet.returnClick = true;
					});						
				}
				
				$('.btn_sean').attr('disabled',false);//다시,이전 버튼 풀기					
			}//클릭가능할때만 END				
		});
	},
	result: function(num, resultSlecter, arrayTxt, arrayObj, arraySet){
		rNum = num - 20,
		arraySet.txtCall = arrayObj[rNum][6],
		arraySet.pdCall =  arrayObj[rNum][1],
		arraySet.imgCall =  arrayObj[rNum][2];
		arraySet.backNum = arrayObj[rNum][7];//백넘버
		
        var viewfileName, submitfileName;	           
		(isMobile)?viewfileName = arrayObj[rNum][4] : viewfileName = arrayObj[rNum][5]
		
	 	viewAdress  = arrayObj[rNum][3] + viewfileName; //자세히 보기 URL		 	
		$(resultSlecter+" .btn_pd_link").off().click(function(){window.open(viewAdress);});
		
		$(resultSlecter+' .pdtext .quest').html(arraySet.txtCall);
		$(resultSlecter+' .pdtext .pd strong').text(arraySet.pdCall);
		
		$(resultSlecter+' .pdcar .car').addClass(arraySet.imgCall);			
		$(resultSlecter+' .faq_btn').css('display','none');
		$(resultSlecter+' .result_btn,'+ resultSlecter+' .pdtext .pd').fadeIn(arraySet.aniSpeed);
		$(resultSlecter+' .pdcar .car,'+ resultSlecter+' .pdtext .quest').animate({'opacity':'1.0'},arraySet.aniSpeed);
		
		if(arraySet.sliderOpt == true){//***** 슬라이더 처리 ******
			var currentHeight = $(resultSlecter).outerHeight();//box 높이
			$(resultSlecter).closest('.faq_slider').css({'height':currentHeight});//box 높이 처리
		}//*****// 슬라이더 처리 ******
	}
}

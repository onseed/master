마케팅용 로딩

  • 이미지 로딩을 시간별로 세분화, 중지시 마케팅에 필요한 만큼 강제 딜레이

실행: $(window).nloading_show();

중지: $(window).nloading_hide();


실행 옵션: $(window).nloading_show({ aniType : 'a', //기본: a, 스케일: b, 슬라이드: c aniDelay : 500 //딜레이 ms 초단위 기본: 500ms (0.5s) txtDelay : 1000 //기본: 1000ms (1s) = 애니메이션 2컷당 1번, 2배 });

중지 옵션: $(window).nloading_hide({ closeTime : 1000 //1초 후 중지: ms 초단위(기본: 0ms) });

var keys = {37: 1, 38: 1, 39: 1, 40: 1};
function preventDefault(e) {
	e = e || window.event;
  	if (e.preventDefault){
  		e.preventDefault();
  	}
  	e.returnValue = false
}
function preventDefaultForScrollKeys(e) {
	if (keys[e.keyCode]) {
		preventDefault(e);
  		return false;
  	}
}
function disableScrollOnly() {//console.log("스크롤 막기");
	if (window.addEventListener){ // older FF
		window.addEventListener('DOMMouseScroll', preventDefault, false);
	}
	window.addEventListener('wheel', preventDefault, {passive:false}); //modern standard 20190917 chrome 패치
	window.ontouchmove  = preventDefault; // mobile
	document.onkeydown  = preventDefaultForScrollKeys;
}
function enableScrollOnly() {//console.log("스크롤 풀기");
	if (window.removeEventListener){
		window.removeEventListener('DOMMouseScroll', preventDefault, false);
	}
	window.removeEventListener('wheel', preventDefault, {passive:false}); //modern standard 20190917 chrome 추가
	window.ontouchmove = null;
	document.onkeydown = null;
}

//#################### nloading_show ####################
nloading_show : function(opt) {
	var defaults = {
			aniType : 'a', //기본: a, 스케일: b,  슬라이드: c
			aniDelay : 500,
			txtDelay : 1000
		},
		opt = $.extend(defaults, opt);
	
	return this.each(function(){
		var aniInDelay = opt.aniDelay,
			aniType = opt.aniType,
			txtInDelay = opt.txtDelay;
		
		if ($.contains($("body")[0], $("div.ui_loading_sec")[0])) {// 이미 딤처리가 되어 있으면 처리하지 않음
			return;
		}
		
		var $loadingSec = $('
'),
			$el		= $('
마케팅용
이미지 로딩
잠시만 기다려주세요
'),
			width	= 0,
			total	= 0,
			left		= 0;
		
		$loadingSec.append($el).appendTo('body');
		width = $('.loading_img').width();
		total = width * -6; //애니컷은 모두 6컷
		
		if(aniType=='b'){//스케일
			$('.loading_img').css("animation","scaleMv "+aniInDelay+"ms infinite");
		}else if(aniType=='c'){//슬라이드
			$('.loading_img').css("animation","transMv "+aniInDelay+"ms infinite");
		}
		
		var txtNum = 0;
		var txtPrevNum = 0;
		function txtTimer() {
			txtNum += 1;
			txtPrevNum = txtNum-1; 
			if(txtNum == 4) { txtNum = 0; txtPrevNum = 3;}
			
			$('.loading_text_box .svtxt').removeClass('txt'+txtPrevNum);
			$('.loading_text_box .svtxt').addClass('txt'+txtNum);
			
			if($.contains($("body")[0], $("div.ui_loading_sec")[0])){
				setTimeout(txtTimer, txtInDelay); //텍스트 애니
			}			
		}
		
		function timer() {
			left -= width;
			(left == total) ? left = 0 : '';
			
			$('.loading_img').css('background-position', left );
			if($.contains($("body")[0], $("div.ui_loading_sec")[0])){
				setTimeout(timer, aniInDelay); //아이콘 애니
			}
			
			setTimeout(function(){
				$('.loading_dim').stop().animate({
					opacity : 0.7
				},200);
			},0);
		}
		
		if($.contains($("body")[0], $("div.ui_loading_sec")[0])){
			setTimeout(timer, aniInDelay);
			setTimeout(txtTimer, txtInDelay); //텍스트 애니
		}
		
		//if(SSCA.isMobile == true){//plugin 적용시
		if($("html").hasClass("mobile")){
			$('html').css('overflow','hidden');
		}
		
		disableScrollOnly();
	});
},
nloading_hide : function(opt) {
	var defaults = {
			closeTime : 0,
		},
		opt = $.extend(defaults, opt);
	
	return this.each(function(){
		var closeTime = opt.closeTime;
		
			setTimeout(function(){
				$('.ui_loading_sec').stop().animate({
					opacity : 0
				},200,function(){
					$('.ui_loading_sec').remove();
				});
				
				//if(SSCA.isMobile == true){//plugin 적용시
				if($("html").hasClass("mobile")){
					$('html').removeAttr('style');
				}
				enableScrollOnly();				
			}, closeTime);
		
		});
	}
//#################### nloading_show ####################
});

데스크탑 기준: 0 / 0 frame

    
var timerInterval,
bgAni = {
	'play' : function(el , fps, col, totalframe){ //el:클래스명 , fps: 초당 프레임, col: 라인당 프레임수, totalframe: 전체 프레임
		
      var width = 0, height= 0,
      total = 0,
      top = 0,
      left = 0,
      xIndex = 0,
      yIndex = 0,
      frameSnd = fps;	        	            

      var frameCol = col, //1라인(행)내 프레임 수
      frameRow = Math.ceil(totalframe/col),//행 갯수 (총프레임수/1행 프레임)	
      frameTotal = totalframe;//총 프레임수
      console.log("frameCol: "+ frameCol + ", frameRow: "+frameRow+", frameTotal: "+frameTotal);
      
      width = $('.'+el).width();
      height = $('.'+el).height();
      total = width * - frameCol;
      frameSnd = Math.floor(1000/frameSnd);

      frameRow = frameRow -1;
      	
      function timer() {
          left -= width;
          xIndex ++;//console.log('left: '+ left + ', top: '+ top +', xIndex: '+ xIndex +', yIndex: '+ yIndex);
        	 if(xIndex == frameTotal){//01. 전체 프레임 완료 후 리셋
        		xIndex = 0;
      		yIndex = 0;
      		top = 0;
      		left = 0;
      		clearTimeout(timerInterval);
      	}
          if(left <= total){//02. 1줄 완료 후  left 리셋//console.log('<= total');
          	left = 0;
          	yIndex ++;
          	if(yIndex==0){
          		top = 0;	
          	}else{
          		top -= height;	
          	}
          }	
          $('.loading_img').css('background-position-x', left);
          $('.loading_img').css('background-position-y', top);
          timerInterval  = setTimeout(timer, frameSnd);//1번 이동
      }						   
      timer();
	},
	'stop' : function(){
		clearTimeout(timerInterval);
	}
}

var mycarCell = {
		basic : function(array){//********** ********** ********** 1. 기본 정보
			$('#maker').text(array.maker); //제조사명 20190403
			$('#name').text(array.name); //차량명
			$('#number').text(array.num); //차량번호
			$('#trim').text(array.trim); //모델명
			$('#image').html('img src="'+array.imgUrl+'" alt="'+array.name+'"/'); //차량 이미지 경로
			$('#price').text(array.price); //출고가
			$('#date').text(array.date); //출시일자				
		},
		priceInfo : function(array, num){//********** ********** ********** 2. 가격 정보
			var disanceDefault = num;
			var maxUnit = array.distancePrice.length;
			
			var txtDistance = array.distancePrice[disanceDefault][0];
				 txtDistance = txtDistance.replace("이상","이상"),
				 txtDistance = txtDistance.replace("미만","미만");
			$('#disance').html(txtDistance); //주행거리 20190128 html태그처리					 
			//$('#disance').text(array.distancePrice[disanceDefault][0]); ////기존 text 처리 삭제
							
			$('#min_price').text(array.distancePrice[disanceDefault][1]); //주행별 최소가
			$('#max_price').text(array.distancePrice[disanceDefault][2]); //주행별 최대가
			
			$('#ui_track').slider({//슬라이더 작동
			    min: 1,
			    range: "min",
			    step: 0.01,
			    max: maxUnit,
			    value: disanceDefault,
			  	animate:50,
			  	orientation: "horizontal",		
			    slide: function( event, ui ) {
			    	var mathNum = Math.round(ui.value);
			    	var gridNum  = mathNum -1;
			    	
					var txtDistance = array.distancePrice[gridNum][0];
				 		txtDistance = txtDistance.replace("이상","이상"),				    	
				 		txtDistance = txtDistance.replace("미만","미만");
			    	$('#disance').html(txtDistance); //주행거리 20190128 html태그처리
			    	//$('#disance').text(array.distancePrice[gridNum][0]); //기존 text 처리 삭제
			    	
			    	$('#min_price').text(array.distancePrice[gridNum][1]);
			    	$('#max_price').text(array.distancePrice[gridNum][2]);	    	
			    },
			    change:function( event, ui ){

			    },
			    stop: function( event, ui ) {
			    	$(this).slider('value',Math.round(ui.value));
			    }
			});	
		},
		papular: function(array, info, num){//********** ********** ********** 3. 인기도  
			var papular = num;
			if(papular == 'undefined' || papular == null ){
				papular = array.papular  -1;
			}
			$('#step_tit').html(''+info[papular][0]+'');
			$('#step_dcs').text(info[papular][1]);
			$('#star_graph').empty();
			var gridLeng = etcInfo.length;
			for(i=0; i < gridLeng; i++){				
				if(i <= papular){
					$('#star_graph').append('');
				}else{
					$('#star_graph').append('');
				}
			}				
		},
		rate: function(array, label){//********** ********** ********** 4. 잔가율
			var rateLeng = array.rate.length;
			$('#bar_graph').empty();
			for(i=0; i < rateLeng; i++){
				var barHeight = 100 - array.rate[i][1];
				//console.log("잔존가치: "+barHeight + "%");
									
				//20190403 마지막 3년후 기준으로 높이가 100%일때 상대값으로 높이처리 -start
				var lastBarValue = 100 - array.rate[rateLeng-1][1]; 
				barHeight =  Math.round(100*barHeight / lastBarValue);
				//20190403 마지막 3년후 기준으로 높이가 100%일때 상대값으로 높이처리 -end
				
				var defaultPrice = comRemove(array.rate[0][0]);
				var minusPrice = comRemove(array.rate[i][0]);
				var ratePrice = comma(defaultPrice - minusPrice);					
				$('#bar_graph').append('
  • '+label[i]+''+array.rate[i][0]+'만원'+array.rate[i][1]+'%-'+ratePrice+'만원
    • ');
				if(i==0){$('.price.num0').remove();}//현재가일때 가격제거
			}
			function comRemove(n){//콤마 제거
				if(typeof n == 'undefined' || n == null || n == ''){return "";}
				var txtNumber = ' ' + n;
				return txtNumber.replace(/(,)/g,"");
			}
			function comma(n) {//콤마 찍기
				return n.toString().replace(/\B(?=(\d{3})+(?!\d))/g, ",");
			}				
		}
}

var loading = {//********** ********** ********** 5. 로딩추가 
	showOverlay:  function(){//show
		var $loadingDim = $('
    '),
			$loadingEl = $('
    '),	
			width       = 0,
			total		= 0,
			left        = 0;
			speed = 100;
		
		if ($.contains($("body")[0], $("div.loading_dim")[0])) {// 딤처리 있으면 return
			return;
		}
		$loadingDim.append($loadingEl).appendTo('body');
		
        	width = $('.loading_el').width();
        	total = width * -10;
        	function timer() {
        		left -= width;
        		(left <= total) ? left = 0 : '';
        		$('.loading_el').css('background-position', left );
        		setTimeout(timer, speed);
        	}
        	if($.contains($("body")[0], $("div.loading_dim")[0])){
        		setTimeout(timer, speed);
        	}
	},
	hideOverlay :  function(){//hide
		$('.loading_dim').fadeOut(200,function(){
			$('.loading_dim').remove();	
		});
	}
}

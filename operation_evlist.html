		function appendLink(id, index1, index2){
			var urlIdtxt;

				urlIdtxt = index1;
			}else{//운영계
				urlIdtxt = index2;
			}

				$(location).attr('href',id[1]);
				e.preventDefault();

신청");

					$(this).closest('.overbox').animate({'top': '117'},300,function(){

					});
	            });				
			}
		}

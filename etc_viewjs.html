3. 조건문: v-if

seen 

	

		
		seen
	



var app3 = new Vue({
	el:'.app-3',
	data:{
		seen:true
	}
});
$(window).load(function(){
	$(".btn_app3").click(function(){
		if(app3.seen == true){
			app3.seen = false;
			$(this).text("show");
		}else{
			app3.seen = true;
			$(this).text("hide");
		}
	});
});

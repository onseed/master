금액 애니메이션

구분 클래스: dot

$.fn.countNum = function(num){//금액 애니메이션
	var $this = $(this),
		$this_value = $this.text().replace(/[^0-9]/g,'');
		$this.text('0');            	
		$this.stop().prop('Counter',0).animate({
			Counter: $this_value
		}, {duration: 900, easing: 'swing',
			step: function(now) {
				if( $this.hasClass('dot') ){ $this.text(dot(Math.ceil(now)));}else{$this.text(comma(Math.ceil(now)));}            							
			}
		});
}            		
function dot(n) {//퍼센트 점 찍기
	return n.toString().replace(/\B(?=(\d{1})+(?!\d))/g, ".");
}
function comma(n) {//금액 콤마
	return n.toString().replace(/\B(?=(\d{3})+(?!\d))/g, ",");
}

setTimeout 과 setIntrval 타이머

  • setTimeout과 setIntrval을 이용한 간단한 타이머와 시간차 비교
  • 스크립트 타이머는 브라우저 성능과 환경에 따라 갱신하는 큐값이 달라 부정확함
  • 타이머수를 줄이고 타이머내 이벤트를 합칠것

실행: TimerTimeout(100); //100은 END숫자

실행: TimerInterval(100); //100은 END숫자

시작시간: 0
setTimeout: 0 > 끝난시간: 0
setInterval: 0 > 끝난시간: 0
정수를 입력하세요

/******** setTimeout 타이머 ********/
function TimerTimeout(num){//num은 END숫자	
	var txtNum = 0;// 초기값
	function timer() {
		txtNum += 1; //1씩 증가
		console.log(txtNum); //이벤트 실행
		
		if(txtNum == num){//타이머 END 
			console.log("END: "+txtNum);
			return false
		}else{
			setTimeout(timer, 0); //타이머 작동, 숫자로 시간조절
		}
	}
	timer();
}
TimerTimeout(100); //100은 END숫자

/******** setInterval 타이머 ********/
function TimerInterval(num){//TimerAni정의: 객체명, num은 END숫자	
	var txtNum = 0;// 초기값
	var intInterval = setInterval(function(){
		txtNum += 1; //1씩 증가 
		console.log(txtNum); //이벤트 실행
		
		if(txtNum == num){//END 
			console.log("END: "+txtNum);
			clearInterval(intInterval);
			return false
		}		
	},0);
}
TimerInterval(100); //100은 END숫자
